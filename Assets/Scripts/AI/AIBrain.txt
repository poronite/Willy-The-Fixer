	fallback
		tree("IsHiding")
		tree("HideFromPlayer")
		tree("GoDestroy")

tree("IsHiding")
	fallback

tree("HideFromPlayer")
	sequence
		HasSeenPlayer
		race
			HideFromPlayer
			sequence
				Wait 10.0
				StopRunningAway
				LookBehind
				while
					not IsSeeing
					sequence
						Wait 15.0
						ForgetPlayer
				

tree("GoDestroy")
	fallback
		GoToTarget
		repeat
			HasReachedTarget
		ReadyDestroy
		not
			Wait 5.0
		DestroyTarget
